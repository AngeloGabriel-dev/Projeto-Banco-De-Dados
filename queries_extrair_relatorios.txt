inner join disciplina D
where H.id_disciplina = D.id_disciplina and H.RA = <<RA do aluno>> 

2)
select id_disciplina, nome, semestre, ano from historico_professor H
inner join disciplina D
where H.id_disciplina = D.id_disciplina and H.id = <<id do professor>>

3)
select * from aluno A
inner join curso C
where A.creditos_totais = C.creditos 

group by 

inner join historico_aluno H
where semestre = <semestre>

5)

inner join grupo_tcc G
